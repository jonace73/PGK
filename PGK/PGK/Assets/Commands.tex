%========== Replace string ==========
\newcommand{\RepAllStr}[1]{%
  \saveexpandmode\noexpandarg
  \def\tempstring{#1}%
  \RepStr{\tempstring}{ff}{\ff}[\tempstring]%
  \RepStr{\tempstring}{fl}{\fl}[\tempstring]%
  \RepStr{\tempstring}{fi}{\fI}[\tempstring]%
  \tempstring
  \restoreexpandmode
}

\newcommand*{\RepStr}{%
  \expandafter\StrSubstitute\expandafter
}
